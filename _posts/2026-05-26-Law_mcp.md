## 1. 개요

- 본 프로젝트는 디지털 포렌식 조사 과정에서 요구되는 엄격한 법적 신뢰성 (무결성 및 증거능력)을 확보하기 위해, AI 모델과 실시간 법률 및 판례 데이터베이스를 연동하는 Law mcp 서버를 구축하는 것을 목적으로 한다. 

1. 위법 수집 증거 방지를 위한 실시간 가이드라인
    - 디지털 데이터는 압수, 수색 시 영장 주의를 위반하기 쉽다. mcp를 통해 AI에게 최신 판례 기준을 실시간으로 참조하게 함으로써, "현재 수사/분석 방식이 독수독과(毒樹毒果) 원칙에 위배되지 않는지" 즉각적으로 체크하는 가이드를 제공하기 위함이다.

2. 연쇄 보관 및 절차의 법적 검증
    디지털 증거가 법정까지 도달하는 과정에서 무결성이 유지되었음을 입증해야 합니다. 각 단계별 작성 문서나 로그가 법적 요건을 충족하는지 MCP 서버를 통해 대법원 가이드라인과 실시간 비교하고 검증한다.

3. 포렌식 분석 결과와 관련 판례의 자동 매칭
    어떤 시스템에서 EXIF 데이터 조직이나 특정 **아티팩트** 삭제 흔적이 발견되었을 경우, 이와 유사한 행위가 재판에서 '증거인멸죄'나 '업무방해죄'로 인정된 구체적인 판례 자료를 mcp가 즉시 AI에게 수급해준다. (공생관계) 이로 인해 수사관은 기술적 분석 결과와 법적 검토를 동시에 끝낼 수 있다. 

* **프로젝트명:** Law mcp (디지털 포렌식 연계)
* **주요 목표:** 법률 정보 검색 및 판례 데이터 인덱싱 서버 구축


- 본 프로젝트의 사이트 이름은 '렉시스'로 명시하며, 이유는 하기와 같다. 
1. 법률 데이터 분야의 상징
    세계에서 가장 유명한 글로벌 법률 및 판례 정보 포털 서비스의 명칭이 '레지스네지스'이다. 
    법률, 포렌식 업계 종사자들에게 Lexis는 전세계 모든 법과 판례가 모여있는 법률 검색의 대명사와 같은 셈이다. 

    // 법률 정보를 전문적으로 다루는 시스템

2. 어원적 의미 : 그리스어
    고대 그리스어에서 유래된 단어로, 언어학에서 '특정 언어의 모든 단어를 총합한 것, 또는 사전' 을 뜻한다. 
    즉, '법률 데이터와 단어를 모두 모아놓은 사전같은 프로그램' 을 의미한다.

  // ── 검색 실행 ────────────────────────────────────────────────────────────
  async function doSearch() {
    const q = document.getElementById('search-input').value.trim();
    if (!q) return;

    const panel = document.getElementById('results-panel');
    const body  = document.getElementById('results-body');
    const count = document.getElementById('results-count');
    const btn   = document.getElementById('search-btn');

    panel.classList.add('open');
    btn.disabled = true;
    body.innerHTML = `<div class="results-loading"><span class="spinner"></span>검색 중...</div>`;
    count.textContent = '';

    try {
      const url = mode === 'case'
        ? `${API}/api/case/search?query=${encodeURIComponent(q)}&display=8`
        : `${API}/api/law/search?query=${encodeURIComponent(q)}&display=8`;

      const res  = await fetch(url);
      if (!res.ok) throw new Error(`HTTP ${res.status}`);
      const data = await res.json();

      count.textContent = `총 ${data.total.toLocaleString()}건`;

      if (!data.items.length) {
        body.innerHTML = `<div class="results-loading">검색 결과가 없습니다.</div>`;
        updateCards([], q);
        return;
      }

      body.innerHTML = data.items.map(
        item => mode === 'case' ? renderCaseResult(item) : renderLawResult(item)
      ).join('');

      updateCards(data.items, q);
    } catch(e) {
      body.innerHTML = `<div class="results-error">
        ⚠ 서버에 연결할 수 없습니다.<br>
        <small>VSCode 터미널에서 <b>uvicorn main:app --reload</b> 를 실행했는지 확인하세요.</small>
      </div>`;
    } finally {
      btn.disabled = false;
    }
  }

      const display = target >= 10000 ? Math.round(target/10000) : target;
      const suffix  = target >= 10000 ? '만+' : '+';
      let start = 0;

        en.target.textContent = Math.floor(e*display).toLocaleString()+(p<1?'':suffix);
        if (p<1) requestAnimationFrame(step);
      };
      requestAnimationFrame(step);
      observer.unobserve(en.target);
    });

  function formatBytes(b) {
    if (b < 1024) return b + ' B';
    if (b < 1048576) return (b/1024).toFixed(2) + ' KB';
    if (b < 1073741824) return (b/1048576).toFixed(2) + ' MB';
    return (b/1073741824).toFixed(2) + ' GB';

  function renderMagicInfo(u8) {

  // ── 3. 해시값 ────────────────────────────────────────────────────────────
  async function renderHashes(buf) {
    const algos = ['SHA-1','SHA-256','SHA-512'];
    const results = await Promise.all(algos.map(a => crypto.subtle.digest(a, buf)));
    const hashes  = results.map(r => Array.from(new Uint8Array(r)).map(b=>b.toString(16).padStart(2,'0')).join(''));

    // MD5는 WebCrypto 미지원 — 간이 구현
    const md5val = md5(buf);

    const all = [['MD5', md5val], ['SHA-1', hashes[0]], ['SHA-256', hashes[1]], ['SHA-512', hashes[2]]];
    document.getElementById('hash-rows').innerHTML = all.map(([algo, val]) => `

      </div>`).join('');
  }

  // 간이 MD5 구현
  function md5(arrayBuffer) {
    const b = new Uint8Array(arrayBuffer);
    let i, j;
    const T = new Uint32Array(64);
    for (i = 0; i < 64; i++) T[i] = Math.floor(Math.abs(Math.sin(i+1)) * 0x100000000) >>> 0;
    const s = [7,12,17,22,7,12,17,22,7,12,17,22,7,12,17,22,5,9,14,20,5,9,14,20,5,9,14,20,5,9,14,20,4,11,16,23,4,11,16,23,4,11,16,23,4,11,16,23,6,10,15,21,6,10,15,21,6,10,15,21,6,10,15,21];
    const msgLen = b.length;
    const bitLen = msgLen * 8;
    const padLen = (msgLen % 64 < 56) ? 56 - msgLen % 64 : 120 - msgLen % 64;
    const padded = new Uint8Array(msgLen + padLen + 8);
    padded.set(b);
    padded[msgLen] = 0x80;
    for (i = 0; i < 8; i++) padded[msgLen + padLen + i] = (bitLen / Math.pow(2, i*8)) & 0xFF;
    let a0=0x67452301,b0=0xEFCDAB89,c0=0x98BADCFE,d0=0x10325476;
    for (i = 0; i < padded.length; i += 64) {
      const M = new Uint32Array(16);
      for (j = 0; j < 16; j++) M[j] = padded[i+j*4]|(padded[i+j*4+1]<<8)|(padded[i+j*4+2]<<16)|(padded[i+j*4+3]<<24);
      let A=a0,B=b0,C=c0,D=d0,F,g;
      for (j = 0; j < 64; j++) {
        if (j<16){F=(B&C)|(~B&D);g=j;}
        else if(j<32){F=(D&B)|(~D&C);g=(5*j+1)%16;}
        else if(j<48){F=B^C^D;g=(3*j+5)%16;}
        else{F=C^(B|(~D));g=(7*j)%16;}
        F=(F+A+T[j]+M[g])>>>0;
        A=D;D=C;C=B;B=(B+((F<<s[j])|(F>>>(32-s[j]))))>>>0;
      }
      a0=(a0+A)>>>0;b0=(b0+B)>>>0;c0=(c0+C)>>>0;d0=(d0+D)>>>0;
    }
    return [a0,b0,c0,d0].map(v=>[v&0xFF,(v>>8)&0xFF,(v>>16)&0xFF,(v>>24)&0xFF].map(b=>b.toString(16).padStart(2,'0')).join('')).join('');
  }

  // ── 4. 엔트로피 분석 ─────────────────────────────────────────────────────
  function calcEntropy(bytes) {
    const freq = new Float64Array(256);
    for (const b of bytes) freq[b]++;
    let H = 0;
    for (const f of freq) {
      if (f === 0) continue;
      const p = f / bytes.length;
      H -= p * Math.log2(p);
    }
    return H;
  }

  // ── 5. 문자열 추출 ───────────────────────────────────────────────────────
  function renderStrings(u8) {
    const MIN_LEN = 4;
    const results = [];
    let cur = '';
    let start = 0;
    for (let i = 0; i <= u8.length; i++) {
      const c = u8[i];
      if (c >= 0x20 && c <= 0x7E) {
        if (cur.length === 0) start = i;
        cur += String.fromCharCode(c);
      } else {
        if (cur.length >= MIN_LEN) results.push([start, cur]);
        cur = '';
        if (results.length >= 200) break;
      }
    }
    const el = document.getElementById('strings-list');
    if (!results.length) { el.innerHTML = '<div style="color:var(--muted);font-size:.8rem;padding:8px 0">추출된 문자열 없음</div>'; return; }
    el.innerHTML = results.map(([off, s]) =>
      `<div class="string-item"><span class="string-offset">0x${off.toString(16).padStart(8,'0')}</span><span class="string-val">${s.replace(/</g,'&lt;')}</span></div>`
    ).join('');
  }

  function hxdRenderRows() {
    const u8    = HXD.buf;
    const start = HXD.viewStart;
    const end   = Math.min(start + HXD.PAGE, u8.length);
    const lines = [];

    for (let i = start - (start % 16); i < end; i += 16) {
      const rowOff = i;
      let cells = '', ascs = '';
      for (let j = 0; j < 16; j++) {
        const off = rowOff + j;
        const b   = u8[off];
        const hex = (off < u8.length) ? b.toString(16).toUpperCase().padStart(2,'0') : '  ';
        const cls = off >= u8.length ? 'c-null' :
                    b === 0          ? 'c-null' :
                    b < 0x20         ? 'c-ctrl' :
                    b < 0x7F         ? 'c-print' :
                    b < 0xA0         ? 'c-ctrl' : 'c-high';
        const selCls   = hxdInSel(off) ? ' c-sel' : '';
        const srchCls  = hxdInSearch(off);
        const editCls  = (HXD.editMode && off === HXD.editOff) ? ' c-edit' : '';
        if (j === 8) cells += '<div class="hxd-spacer"></div>';
        cells += `<div class="hxd-cell ${cls}${selCls}${srchCls}${editCls}" data-off="${off}" onmousedown="hxdMouseDown(event,${off})" onmouseover="hxdMouseOver(event,${off})" onmouseup="hxdMouseUp(event,${off})" onclick="hxdClick(${off})">${hex}</div>`;
        const ch = (off < u8.length && b >= 0x20 && b < 0x7F) ? String.fromCharCode(b).replace(/&/g,'&amp;').replace(/</g,'&lt;') : '.';
        const aCls = (off < u8.length && b >= 0x20 && b < 0x7F) ? 'a-print' : '';
        const aSelCls = hxdInSel(off) ? ' a-sel' : (hxdInSearch(off) ? hxdInSearch(off).replace('c-','a-') : '');
        ascs += `<span class="hxd-asc ${aCls}${aSelCls}" data-off="${off}">${ch}</span>`;
      }
      const offStr = rowOff.toString(16).toUpperCase().padStart(8,'0');
      lines.push(`<div class="hxd-row${hxdInSel(rowOff)||hxdInSel(rowOff+15)?' row-selected':''}" data-rowoff="${rowOff}"><div class="hxd-offset">${offStr}</div><div class="hxd-bytes">${cells}</div><div class="hxd-ascii-col">${ascs}</div></div>`);
    }
    document.getElementById('hxd-rows').innerHTML = lines.join('');
    document.getElementById('hxd-st-range').textContent = `${start.toString(16).toUpperCase().padStart(8,'0')} – ${Math.min(end,u8.length).toString(16).toUpperCase().padStart(8,'0')}`;
  }

  function hxdInSel(off) {
    if (HXD.selStart < 0 || off < 0 || off >= HXD.buf.length) return false;
    const lo = Math.min(HXD.selStart, HXD.selEnd);
    const hi = Math.max(HXD.selStart, HXD.selEnd);
    return off >= lo && off <= hi;
  }

  function hxdInSearch(off) {
    if (!HXD.searchMatches.length) return '';
    for (let m = 0; m < HXD.searchMatches.length; m++) {
      const [ms, me] = HXD.searchMatches[m];
      if (off >= ms && off <= me) return m === HXD.searchCur ? ' c-search-cur' : ' c-search';
    }
    return '';
  }

function hxdBindScroll() {
  const body = document.getElementById('hxd-body');
  if (!body) return; // 💡 엘리먼트가 없으면 매칭 연산 패스!
  
  body.onscroll = () => {
    if (!HXD.buf) return; // 파일 데이터 버퍼가 없으면 연산 안 함
    const rowH = 20;
    const rowIdx = Math.floor(body.scrollTop / rowH);
    const newStart = rowIdx * 16;
    if (Math.abs(newStart - HXD.viewStart) > 32) {
      HXD.viewStart = Math.max(0, newStart - 32);
      hxdRenderRows();
    }
  };
}

    if (HXD.selStart >= 0) {
      const len = hi - lo + 1;

  // ── 선택 영역 통계 패널 ──────────────────────────────────────────────────
  function hxdUpdateSelPanel() {
    const panel = document.getElementById('hxd-sel-panel');
    if (HXD.selStart < 0 || HXD.selStart === HXD.selEnd) { panel.classList.remove('show'); return; }
    const lo  = Math.min(HXD.selStart, HXD.selEnd);
    const hi  = Math.max(HXD.selStart, HXD.selEnd);
    const sel = HXD.buf.slice(lo, hi + 1);

    // 정수 해석 (리틀 엔디안)
    const dv = new DataView(sel.buffer, sel.byteOffset, sel.byteLength);
    const interpretations = [];
    if (sel.length >= 1) interpretations.push(['UINT8',  sel[0]],               ['INT8',   new DataView(sel.buffer, sel.byteOffset, 1).getInt8(0)]);
    if (sel.length >= 2) interpretations.push(['UINT16-LE', dv.getUint16(0,true)], ['UINT16-BE', dv.getUint16(0,false)]);
    if (sel.length >= 4) interpretations.push(['UINT32-LE', dv.getUint32(0,true)], ['INT32-LE',  dv.getInt32(0,true)],  ['FLOAT32', dv.getFloat32(0,true).toFixed(6)]);
    if (sel.length >= 8) interpretations.push(['INT64-LE',  BigInt.asIntN(64, dv.getBigUint64(0,true)).toString()], ['FLOAT64', dv.getFloat64(0,true).toExponential(4)]);

    // XOR / SUM
    let xorVal = 0, sumVal = 0;
    for (const b of sel) { xorVal ^= b; sumVal += b; }
    interpretations.push(['XOR', '0x' + xorVal.toString(16).toUpperCase().padStart(2,'0')], ['SUM', sumVal + ` (0x${sumVal.toString(16).toUpperCase()})`]);

    // 텍스트 해석
    const text = Array.from(sel).map(b => b >= 0x20 && b < 0x7F ? String.fromCharCode(b) : '.').join('');
    interpretations.push(['TEXT', text.length > 40 ? text.slice(0,40)+'…' : text]);

    document.getElementById('hxd-sel-grid').innerHTML = interpretations.map(([k,v]) =>
      `<div class="hxd-sel-item"><div class="hxd-sel-label">${k}</div><div class="hxd-sel-value">${String(v).replace(/</g,'&lt;')}</div></div>`
    ).join('');
    panel.classList.add('show');
  }

  // ── 검색 ─────────────────────────────────────────────────────────────────
  function hxdSearch(dir) {
    const raw   = document.getElementById('hxd-sinput').value.trim();
    const stype = document.getElementById('hxd-stype').value;
    if (!raw || !HXD.buf) return;

    let needle;
    if (stype === 'hex') {
      const parts = raw.replace(/\s+/g,' ').split(' ').filter(Boolean);
      needle = parts.map(p => parseInt(p, 16)).filter(n => !isNaN(n));
    } else {
      needle = Array.from(raw).map(c => c.charCodeAt(0));
    }
    if (!needle.length) return;

    // 전체 매치 목록 갱신
    HXD.searchMatches = [];
    for (let i = 0; i <= HXD.buf.length - needle.length; i++) {
      if (needle.every((b, j) => HXD.buf[i+j] === b)) {
        HXD.searchMatches.push([i, i + needle.length - 1]);
      }
    }

    if (!HXD.searchMatches.length) {
      document.getElementById('hxd-match-info').textContent = '없음';
      HXD.searchCur = -1;
      hxdRenderRows();
      return;
    }

    // 현재 커서 기준 다음/이전
    const cur = HXD.searchCur < 0 ? 0 : HXD.searchCur;
    HXD.searchCur = (cur + dir + HXD.searchMatches.length) % HXD.searchMatches.length;
    document.getElementById('hxd-match-info').textContent = `${HXD.searchCur+1} / ${HXD.searchMatches.length}`;

    const [ms, me] = HXD.searchMatches[HXD.searchCur];
    HXD.selStart = ms; HXD.selEnd = me;
    HXD.viewStart = Math.max(0, ms - 64);

  function hxdCopySel(fmt) {
    if (HXD.selStart < 0) { showToast('먼저 범위를 선택하세요'); return; }
    const lo  = Math.min(HXD.selStart, HXD.selEnd);
    const hi  = Math.max(HXD.selStart, HXD.selEnd);
    const sel = HXD.buf.slice(lo, hi + 1);
    let out;
    if (fmt === 'hex') out = Array.from(sel).map(b => b.toString(16).toUpperCase().padStart(2,'0')).join(' ');
    else if (fmt === 'dec') out = Array.from(sel).map(b => b).join(', ');
    else out = Array.from(sel).map(b => b >= 0x20 && b < 0x7F ? String.fromCharCode(b) : '.').join('');
    copyText(out);
  }

# def _text(el: ET.Element, tag: str, default: str = "") -> str:
#    node = el.find(tag)
#    return (node.text or default).strip() if node is not None else default

def _text(el: ET.Element, tag: str, default: str = "") -> str:
    node = el.find(tag)
    if node is None:
        return default
    
    # .text 대신 itertext()를 사용하여 CDATA 내부 텍스트까지 안전하게 병합
    text_content = "".join(node.itertext())

    # 만약 결과가 없다면 기본값 반환, 있다면 공백 제거 후 반환
    return text_content.strip() if text_content else default

async def law_get(path: str, params: dict) -> ET.Element:
    params = {"OC": API_KEY, "type": "XML", **params}
    async with httpx.AsyncClient(timeout=15) as client:
        resp = await client.get(f"{BASE}/{path}", params=params)
    if resp.status_code != 200:
        raise HTTPException(status_code=502, detail="법령 API 오류")
    return ET.fromstring(resp.text)

@app.get("/api/law/search")

    query:   str = Query(..., description="검색 키워드"),
    page:    int = Query(1,  ge=1),
    display: int = Query(10, ge=1, le=20),
):

        "target": "law",
        "query": query,
        "page": page,
        "display": display,
    })

    items = []
    for item in root.findall(".//law"):
        items.append({
            "id":        _text(item, "법령ID"),
            "name":      _text(item, "법령명한글"),
            "abbr":      _text(item, "법령명약칭"),
            "ministry":  _text(item, "소관부처명"),
            "promulgated": _text(item, "공포일자"),
            "type":      _text(item, "법령구분명"),
        })

    return {
        "total":   int(_text(root, "totalCnt", "0")),
        "page":    page,
        "display": display,
        "items":   items,
    }
